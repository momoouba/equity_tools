## 1. 获取管理人指标

### 基本信息

- **接口地址**: `GET /api/performance/dashboard/manager`
- **接口说明**: 获取管理人指标卡数据

### 请求参数

| 参数 | 类型 | 必填 | 说明 |
|------|------|------|------|
| version | string | 是 | 版本号 |

### 响应数据

```typescript
interface ManagerIndicator {
  fofNum: number;            // 母基金数量
  directNum: number;         // 直投基金数量
  subAmount: number;         // 认缴管理规模（元）
  subAdd: number;            // 认缴规模较上年变动（元）
  paidInAmount: number;      // 实缴管理规模（元）
  paidInAdd: number;         // 实缴规模较上年变动（元）
  disAmount: number;         // 累计分配总额（元）
  disAdd: number;            // 累计分配较上年变动（元）
}
```

### 响应示例

```json
{
  "code": 200,
  "message": "success",
  "data": {
    "fofNum": 4,
    "directNum": 5,
    "subAmount": 33756000000,
    "subAdd": 0,
    "paidInAmount": 21112000000,
    "paidInAdd": -225000000,
    "disAmount": 6076000000,
    "disAdd": 488000000
  },
  "timestamp": 1704067200000
}
```

### SQL参考

```sql
SELECT 
  fof_num,
  direct_num,
  sub_amount,
  sub_add,
  paid_in_amount,
  paid_in_add,
  dis_amount,
  dis_add
FROM b_manage_indicator
WHERE version = ? AND F_DeleteMark = 0
```

## 2. 获取在管产品清单

### 基本信息

- **接口地址**: `GET /api/performance/dashboard/manager/funds`
- **接口说明**: 获取在管产品清单弹窗数据

### 响应数据

```typescript
interface FundProduct {
  fund: string;              // 基金名称
  fundType: string;          // 基金类型
  subAmount: number;         // 认缴规模
  subAdd: number;            // 本年新增认缴
  paidInAmount: number;      // 实缴规模
  paidInAdd: number;         // 本年新增实缴
  disAmount: number;         // 累计分配金额
  disAdd: number;            // 本年新增分配
  setUpDate: string;         // 成立时间
}

interface FundProductListResponse {
  list: FundProduct[];
  summary: {
    subAmountTotal: number;
    paidInAmountTotal: number;
    paidInAddTotal: number;
    disAmountTotal: number;
    disAddTotal: number;
  };
}
```

### SQL参考

```sql
SELECT 
  fund,
  fund_type,
  sub_amount,
  sub_add,
  paid_in_amount,
  paid_in_add,
  dis_amount,
  dis_add,
  set_up_date
FROM b_manage
WHERE version = ? AND F_DeleteMark = 0
ORDER BY fund_type, set_up_date DESC
```

## 3. 获取基金产品指标

### 基本信息

- **接口地址**: `GET /api/performance/dashboard/fund-products`
- **接口说明**: 获取基金产品指标块数据，包括横向基金列表和纵向指标数据

### 响应数据

```typescript
interface FundProductIndicator {
  funds: string[];           // 基金名称列表（横向表头）
  indicators: {
    lpSub: number[];         // 投资人认缴
    paidin: number[];        // 投资人实缴
    distribution: number[];  // 投资人分配
    tvpi: number[];          // TVPI
    dpi: number[];           // DPI
    rvpi: number[];          // RVPI
    nirr: number[];          // NIRR
    subAmount: number[];     // 投资金额/认缴
    invAmount: number[];     // 投资金额/实缴
    exitAmount: number[];    // 退出金额
    girr: number[];          // GIRR
    moc: number[];           // MOC
  };
}
```

### SQL参考

```sql
-- 获取基金列表
SELECT fund 
FROM b_manage
WHERE version = ? AND F_DeleteMark = 0
ORDER BY fund_type, set_up_date ASC;

-- 获取指标数据
SELECT 
  fund,
  lp_sub,
  paidin,
  distribution,
  tvpi,
  dpi,
  rvpi,
  nirr,
  sub_amount,
  inv_amount,
  exit_amount,
  girr,
  moc
FROM b_transaction_indicator
WHERE version = ? AND F_DeleteMark = 0
ORDER BY fund_type, set_up_date ASC
```

## 4. 获取投资人名录

### 基本信息

- **接口地址**: `GET /api/performance/dashboard/funds/:fund/investors`
- **接口说明**: 获取指定基金的投资人名录弹窗数据

### 请求参数

| 参数 | 类型 | 必填 | 说明 |
|------|------|------|------|
| fund | string | 是 | 基金名称，路径参数 |
| version | string | 是 | 版本号，查询参数 |

### 响应数据

```typescript
interface Investor {
  fund: string;                    // 基金名称
  lpType: string;                  // 投资人类型
  lp: string;                      // 投资人名称
  subscriptionAmount: number;      // 认缴金额
  subscriptionRatio: number;       // 认缴比例
  paidin: number;                  // 累计实缴金额
  distribution: number;            // 累计分配金额
  firstDate: string | null;        // 第N次分配时间
  firstAmount: number | null;      // 第N次分配金额
  secondDate: string | null;       // 第N-1次分配时间
  secondAmount: number | null;     // 第N-1次分配金额
  thirdDate: string | null;        // 第N-2次分配时间
  thirdAmount: number | null;      // 第N-2次分配金额
}

interface InvestorListResponse {
  list: Investor[];
  distributionDates: {
    first: string | null;
    second: string | null;
    third: string | null;
  };
  summary: {
    subscriptionAmountTotal: number;
    paidinTotal: number;
    distributionTotal: number;
    firstAmountTotal: number;
    secondAmountTotal: number;
    thirdAmountTotal: number;
  };
}
```

### SQL参考

```sql
SELECT 
  fund,
  lp_type,
  lp,
  subscription_amount,
  subscription_ratio,
  paidin,
  distribution,
  first_date,
  first_amount,
  second_date,
  second_amount,
  third_date,
  third_amount
FROM b_investor_list
WHERE version = ? AND F_DeleteMark = 0 AND fund = ?
```

## 5. 获取基金业绩指标

### 基本信息

- **接口地址**: `GET /api/performance/dashboard/funds/:fund/performance`
- **接口说明**: 获取基金业绩指标及现金流底表弹窗数据

### 响应数据

```typescript
interface FundPerformance {
  indicator: {
    fund: string;
    lpSub: number;
    paidin: number;
    distribution: number;
    tvpi: number;
    dpi: number;
    rvpi: number;
    nirr: number;
  };
  transactions: {
    fund: string;
    lp: string;
    transactionType: string;
    transactionDate: string;
    transactionAmount: number;
  }[];
}
```

### SQL参考

```sql
-- 指标数据
SELECT 
  fund, lp_sub, paidin, distribution, tvpi, dpi, rvpi, nirr
FROM b_transaction_indicator
WHERE version = ? AND F_DeleteMark = 0 AND fund = ?;

-- 交易明细
SELECT 
  fund, lp, transaction_type, transaction_date, transaction_amount
FROM b_transaction
WHERE version = ? AND F_DeleteMark = 0 
  AND fund = ? AND lp IS NOT NULL
ORDER BY fund, transaction_date DESC
```

## 6. 获取基金投资组合

### 基本信息

- **接口地址**: `GET /api/performance/dashboard/funds/:fund/portfolio`
- **接口说明**: 获取基金投资组合明细弹窗数据

### 响应数据

```typescript
interface PortfolioItem {
  fund: string;
  transactionType: string;   // 投资类别：子基金/直投项目
  project: string;           // 项目名称
  firstDate: string;         // 投资时间
  accSub: number;            // 认缴金额累计
  changeSub: number;         // 认缴金额本月变动
  accPaidin: number;         // 实缴金额累计
  changePaidin: number;      // 实缴金额本月变动
  accExit: number;           // 退出金额累计
  changeExit: number;        // 退出金额本月变动
  accReceive: number;        // 回款金额累计
  changeReceive: number;     // 回款金额本月变动
  unrealized: number;        // 未实现价值
  changeUnrealized: number;  // 未实现价值变动
  totalValue: number;        // 总价值
  moc: number;               // MOC
  dpi: number;               // DPI
}

interface FundPortfolioResponse {
  subFunds: {
    list: PortfolioItem[];
    summary: PortfolioSummary;
  };
  directInvestments: {
    list: PortfolioItem[];
    summary: PortfolioSummary;
  };
  totalSummary: PortfolioSummary;
}
```

### SQL参考

```sql
SELECT 
  fund, transaction_type, project, first_date,
  acc_sub, change_sub, acc_paidin, change_paidin,
  acc_exit, change_exit, acc_receive, change_receive,
  unrealized, change_unrealized, total_value, moc, dpi
FROM b_investment
WHERE version = ? AND F_DeleteMark = 0 AND fund = ?
ORDER BY transaction_type, first_date ASC
```

## 8. 获取投资组合指标

### 基本信息

- **接口地址**: `GET /api/performance/dashboard/portfolio`
- **接口说明**: 获取投资组合模块数据

### 响应数据

```typescript
interface PortfolioIndicator {
  // 各基金数据
  funds: {
    name: string;
    subFund: {
      inv: number;       // 投数量
      exit: number;      // 退数量
      sub: number;       // 认缴金额
      exitAmount: number;// 退出金额
      paidin: number;    // 实缴金额
      receive: number;   // 回款金额
    };
    project: {
      inv: number;
      exit: number;
      paidin: number;
      receive: number;
    };
  }[];

  overall: {
    fundInv: number;           // 子基金累计投资数量
    fundInvChange: number;     // 变动
    fundSub: number;           // 子基金累计认缴金额

    projectPaidin: number;     // 直投项目累计投资金额
    // ... 其他字段
  };
}
```

### SQL参考

```sql
-- 各基金数据
SELECT 
  fund, fund_inv, fund_exit, fund_sub, fund_exit_amount,
  fund_paidin, fund_receive, project_inv, project_exit,
  project_paidin, project_receive
FROM b_investment_indicator
WHERE version = ? AND F_DeleteMark = 0;

SELECT *
FROM b_all_indicator
WHERE version = ? AND F_DeleteMark = 0
```

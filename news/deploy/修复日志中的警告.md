## ✅ 修复步骤

### 修复外键约束警告

```bash
# 连接到数据库
sudo mysql -u root -p'Mqdqxygyqy!!!klklsys24678' <<EOF
USE investment_tools;

# 检查外键约束是否存在
SELECT CONSTRAINT_NAME 
FROM INFORMATION_SCHEMA.KEY_COLUMN_USAGE 
WHERE TABLE_SCHEMA = 'investment_tools' 
AND TABLE_NAME = 'news_interface_config' 
AND CONSTRAINT_NAME LIKE '%app_id%';

# 如果存在重复的外键，先删除所有相关外键
SET @constraint_name = NULL;
SELECT CONSTRAINT_NAME INTO @constraint_name
FROM INFORMATION_SCHEMA.KEY_COLUMN_USAGE 
WHERE TABLE_SCHEMA = 'investment_tools' 
AND TABLE_NAME = 'news_interface_config' 
AND REFERENCED_TABLE_NAME = 'applications'
AND COLUMN_NAME = 'app_id'
LIMIT 1;

SET @sql = IF(@constraint_name IS NOT NULL, 
    CONCAT('ALTER TABLE news_interface_config DROP FOREIGN KEY ', @constraint_name), 
    'SELECT "No foreign key found"');
PREPARE stmt FROM @sql;
EXECUTE stmt;
DEALLOCATE PREPARE stmt;

# 重新添加外键约束
ALTER TABLE news_interface_config 
ADD CONSTRAINT fk_news_interface_config_app_id 
FOREIGN KEY (app_id) REFERENCES applications(id) ON DELETE CASCADE;

SELECT '外键约束修复完成' AS result;
EOF
```

## 🚀 一键修复外键约束

```bash
# 修复外键约束问题
sudo mysql -u root -p'Mqdqxygyqy!!!klklsys24678' <<EOF
USE investment_tools;

-- 删除所有可能重复的外键约束
SET FOREIGN_KEY_CHECKS = 0;

-- 查找并删除所有相关外键
SELECT CONCAT('ALTER TABLE news_interface_config DROP FOREIGN KEY ', CONSTRAINT_NAME, ';') AS sql_statement
FROM INFORMATION_SCHEMA.KEY_COLUMN_USAGE 
WHERE TABLE_SCHEMA = 'investment_tools' 
AND TABLE_NAME = 'news_interface_config' 
AND REFERENCED_TABLE_NAME = 'applications'
AND COLUMN_NAME = 'app_id'
INTO OUTFILE '/tmp/drop_fk.sql';

-- 重新添加外键（如果不存在）
ALTER TABLE news_interface_config 
DROP FOREIGN KEY IF EXISTS fk_news_interface_config_app_id;

ALTER TABLE news_interface_config 
ADD CONSTRAINT fk_news_interface_config_app_id 
FOREIGN KEY (app_id) REFERENCES applications(id) ON DELETE CASCADE;

SET FOREIGN_KEY_CHECKS = 1;

SELECT '外键约束修复完成' AS result;
EOF

# 重启应用
pm2 restart newsapp
```
